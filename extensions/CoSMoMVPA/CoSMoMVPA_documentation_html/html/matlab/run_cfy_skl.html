ds=cosmo_synthetic_dataset('ntargets',2,'nchunks',100,'size','big');
partitions=cosmo_oddeven_partitioner(ds);

niter=100;

train=partitions.train_indices{1};
test=partitions.test_indices{1};

y=ds.sa.targets(train);

opt=struct();
opt.partitions=partitions;
opt.classifier=@cosmo_classify_matlabsvm_2class;

t=0;
for k=1:niter
    xs=randn(size(ds.samples));
    s=tic();
    %cosmo_classify_matlabsvm_2class(xs(train,:),y,xs(test,:));
    cosmo_crossvalidation_measure(ds,opt);
    t=t+toc(s);
end

fprintf('t=%.3f',t);
